## 1. Convert CSV to Dataframe for pd

```python
import pandas as pd

# CSV 파일을 pandas DataFrame으로 읽어옵니다.
# <file path>에는 train.csv 같은 실제 CSV 경로를 넣습니다.
train_df = pd.read_csv("<file path>")

# label 컬럼에 들어 있는 문자열 값을 학습에 사용할 숫자 클래스로 바꿉니다.

train_df['label'] = train_df['label'].apply(lambda x : int(x))

display(train_df['label'].value_counts())
```


## 2. Get image path and label list 

```python
from glob import glob
import os

def get_train_data(data_dir):
    # 이미지 파일 경로를 담을 리스트입니다.
    img_path_list = []

    # 각 이미지의 정답 label을 담을 리스트입니다.
    label_list = []

    # data_dir 폴더 안의 모든 jpg 파일 경로를 가져옵니다.
    img_path_list.extend(glob(os.path.join(data_dir, '*.jpg')))

    # 파일명이 숫자라고 가정하고 숫자 순서대로 정렬합니다.
    # 예: 1.jpg, 2.jpg, 10.jpg 순서가 깨지지 않게 하기 위한 처리입니다.
    img_path_list.sort(key=lambda x: int(x.split('/')[-1].split('.')[0]))  

    # CSV에서 읽은 label 컬럼을 label_list에 추가합니다.
    label_list.extend(train_df['label'])

    # 학습 이미지 경로 리스트와 라벨 리스트를 함께 반환합니다.
    return img_path_list, label_list

def get_test_data(data_dir):
    # 테스트 데이터는 정답 label이 없으므로 이미지 경로만 가져옵니다.
    img_path_list = []

    # data_dir 폴더 안의 모든 jpg 파일 경로를 가져옵니다.
    img_path_list.extend(glob(os.path.join(data_dir, '*.jpg')))

    # 파일명 숫자 기준으로 정렬합니다.
    img_path_list.sort(key=lambda x: int(x.split('/')[-1].split('.')[0]))

    return img_path_list

# 학습 이미지 경로와 라벨을 가져옵니다.
all_img_path, all_label = get_train_data('sign_train')

# 테스트 이미지 경로를 가져옵니다.
test_img_path = get_test_data('sign_test')

# 앞쪽 데이터가 잘 읽혔는지 확인합니다.
print(all_img_path[:10])
print(all_label[:10])
```

## 3. Albumentations 를 활용한 이미지 데이터 전처리 및 변환 

```python
import albumentations as A
from albumentations.pytorch import ToTensorV2

# 학습 데이터에 적용할 이미지 변환입니다.
# Resize -> Normalize -> Tensor 변환 순서로 실행됩니다.
train_transform = A.Compose([
    # 모든 이미지를 64x64 크기로 맞춥니다.
    # CNN은 한 batch 안의 이미지 크기가 같아야 합니다.
    A.Resize(64, 64),

    # 픽셀값을 0~1 범위로 바꾼 뒤 ImageNet 평균/표준편차로 정규화합니다.
    # 사전학습 모델을 쓸 때 자주 쓰는 기본 통계값입니다.
    A.Normalize(mean=(0.485, 0.456, 0.406), std=(0.229, 0.224, 0.225), max_pixel_value=255.0, always_apply=False, p=1.0),

    # numpy 이미지(H, W, C)를 PyTorch tensor(C, H, W)로 변환합니다.
    ToTensorV2()
])

# 검증/테스트 데이터에 적용할 이미지 변환입니다.
# 평가 데이터에는 보통 랜덤 증강을 넣지 않고, 크기 조정과 정규화만 적용합니다.
test_transform = A.Compose([
    A.Resize(64, 64),
    A.Normalize(mean=(0.485, 0.456, 0.406), std=(0.229, 0.224, 0.225), max_pixel_value=255.0, always_apply=False, p=1.0),
    ToTensorV2()
])
```

### 224x224가 아닌 64x64로 변환하는 이유

- 모델의 경량화
    - 224의 경우 이미지는 높은 해상도를 제공하지만, 그만큼 더 많은 계산을 요구 
    - 모델의 파라미터 수가 감소하면 경량화된 모델을 설계할 수 있다. 
- 처리 효율성
    - 더 작은 입력 크기는 연산 속도를 빠르게 하여 모델의 추론(inference) 시간을 단축시킨다. 학습 과정에서도 처리 효율성이 향상되어 더 많은 데이터를 짧은 시간에 학습, 실험을 빠르게 반복할 수 있다. 

### A.Normalize(mean, std)

- 이미지의 픽셀 값을 0과 1사이의 값으로 변환한 다음, 주어진 평균과 표준편차를 사용하여 정규화 

- `mean(A, B, C)` : 정규화 할때 사용할 평균 값. ImageNet데이터 셋에서 사전학습된 모델의 평균값을 사용

- `std(A,B,C)` : 정규화 할때 사용할 표준편차, ImageNet 데이터 셋의 표준편차 값 

- `max_pixel_value=255`, 이미지의 최대 픽셀 값, 일반적으로 이미지의 픽셀 값은 0에서 255 사이의 정수, 이를 255로 지정하여 나눈다. 

- `p` : 이 변환을 항상 적용하도록 설정 

#### 정규화 과정

1. 픽셀 값을 0과 1사이로 변환
    - 0~255사이의 값을 255로 나누어 0과 1사이의 값으로 변환
2. 평균을 빼고 표준편차로 나눔
    - $(픽셀값 - 평균)/ 표준편차$ 공식 사용 
    - 표준에서의 값의 거리 

#### TensorV2

- 이미지를 pytorch tensor형식으로 변환 
- HWC 에서 CHW로 변경하여 pytorch로 사용 가능한 형식으로 맞춤 

## 4. Pytorch 커스텀 데이터 셋 클래스 구현: 이미지 로드 및 변환 

```python
from torch.utils.data import Dataset, DataLoader

# PyTorch Dataset은 __len__과 __getitem__을 구현해야 합니다.
# DataLoader는 이 Dataset을 사용해서 batch를 만들어 줍니다.
class CustomDataset(Dataset):
    def __init__(self, img_paths, labels=None, transform=None):
        # 이미지 파일 경로 목록입니다.
        self.img_paths = img_paths

        # 정답 label 목록입니다.
        # 테스트 데이터처럼 label이 없을 수도 있으므로 기본값은 None입니다.
        self.labels = labels

        # 이미지에 적용할 전처리/증강 함수입니다.
        self.transform = transform

    def __len__(self):
        # Dataset의 전체 데이터 개수를 반환합니다.
        # len(dataset)을 호출하면 이 함수가 실행됩니다.
        return len(self.img_paths)

    def __getitem__(self, idx):
        # idx번째 이미지 경로를 가져옵니다.
        img_path = self.img_paths[idx]

        # OpenCV로 이미지를 읽습니다.
        # cv2.imread는 이미지를 BGR 순서로 읽습니다.
        image = cv2.imread(img_path)

        # 일반적인 RGB 이미지 순서로 변환합니다.
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

        # transform이 있으면 이미지 전처리/증강을 적용합니다.
        if self.transform:
            augmented = self.transform(image=image)
            image = augmented['image']

        # label이 있는 학습/검증 데이터는 image와 label을 함께 반환합니다.
        if self.labels is not None:
            label = self.labels[idx]
            return image, label

        # label이 없는 테스트 데이터는 image만 반환합니다.
        else:
            return image
```

## 7. pytorch를 사용한 간단한 CNN 모델 구현 및 초기화 

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

# nn.Module을 상속해서 직접 CNN 모델을 정의합니다.
class SimpleCNN(nn.Module):
    def __init__(self, num_classes):
        super(SimpleCNN, self).__init__()

        # 첫 번째 convolution layer입니다.
        # 입력 채널 3개(RGB), 출력 채널 16개, 커널 크기 3x3입니다.
        # padding=1을 사용하면 64x64 입력 크기가 convolution 후에도 64x64로 유지됩니다.
        self.conv1 = nn.Conv2d(3, 16, kernel_size=3, padding=1)

        # 두 번째 convolution layer입니다.
        # conv1이 만든 16개 feature map을 입력으로 받아 32개 feature map을 만듭니다.
        # padding=1이므로 가로/세로 크기는 유지됩니다.
        self.conv2 = nn.Conv2d(16, 32, kernel_size=3, padding=1)

        # MaxPool2d는 feature map의 가로/세로 크기를 줄입니다.
        # kernel_size=2, stride=2이면 2x2 영역마다 가장 큰 값 하나만 남겨 크기가 절반이 됩니다.
        # 예: 64x64 -> 32x32, 32x32 -> 16x16
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2, padding=0)

        # 두 번의 pooling 후 텐서 크기는 (32, 16, 16)이 됩니다.
        # Linear layer는 1차원 벡터를 입력으로 받으므로 32 * 16 * 16개 값을 입력으로 사용합니다.
        self.fc1 = nn.Linear(32 * 16 * 16, 64)

        # 최종 분류 layer입니다.
        # 출력 크기는 클래스 개수(num_classes)입니다.
        self.fc2 = nn.Linear(64, num_classes)

    def forward(self, x):
        # 입력 x shape 예: (batch_size, 3, 64, 64)
        # conv1 -> ReLU -> pool 순서로 처리합니다.
        # 결과 shape: (batch_size, 16, 32, 32)
        x = self.pool(F.relu(self.conv1(x)))

        # conv2 -> ReLU -> pool 순서로 처리합니다.
        # 결과 shape: (batch_size, 32, 16, 16)
        x = self.pool(F.relu(self.conv2(x)))

        # Linear layer에 넣기 위해 4차원 텐서를 2차원 텐서로 펼칩니다.
        # (batch_size, 32, 16, 16) -> (batch_size, 32 * 16 * 16)
        # -1은 batch_size를 PyTorch가 자동으로 계산하라는 뜻입니다.
        x = x.view(-1, 32 * 16 * 16)  

        # 첫 번째 fully connected layer를 통과한 뒤 ReLU를 적용합니다.
        x = F.relu(self.fc1(x))

        # 최종 class별 점수(logits)를 출력합니다.
        # CrossEntropyLoss를 사용할 때는 softmax를 직접 적용하지 않습니다.
        x = self.fc2(x)
        return x

# 모델 생성
# label의 고유값 개수를 class 개수로 사용합니다.
num_classes = len(train_df['label'].unique())

# SimpleCNN 모델 객체를 생성합니다.
model = SimpleCNN(num_classes)

# 모델 구조를 출력합니다.
print(model)
```

### 왜 `nn.Linear`의 `in_features`가 `32 * 16 * 16`인가?

입력 이미지는 `A.Resize(64, 64)`를 거쳤기 때문에 모델에 들어갈 때 크기가 `3 x 64 x 64`입니다.

1. 입력 이미지 크기
    - RGB 이미지이므로 채널 수는 `3`입니다.
    - Resize 이후 가로/세로는 `64 x 64`입니다.
    - 따라서 입력 shape은 `(batch_size, 3, 64, 64)`입니다.

2. `conv1 = nn.Conv2d(3, 16, kernel_size=3, padding=1)`
    - 출력 채널 수가 `16`으로 바뀝니다.
    - `kernel_size=3`, `padding=1`, `stride=1`이면 가로/세로 크기는 유지됩니다.
    - 결과 shape은 `(batch_size, 16, 64, 64)`입니다.

3. 첫 번째 `MaxPool2d(kernel_size=2, stride=2)`
    - 가로/세로 크기를 절반으로 줄입니다.
    - `64 x 64`가 `32 x 32`가 됩니다.
    - 결과 shape은 `(batch_size, 16, 32, 32)`입니다.

4. `conv2 = nn.Conv2d(16, 32, kernel_size=3, padding=1)`
    - 출력 채널 수가 `32`로 바뀝니다.
    - padding 때문에 가로/세로 크기는 `32 x 32`로 유지됩니다.
    - 결과 shape은 `(batch_size, 32, 32, 32)`입니다.

5. 두 번째 `MaxPool2d(kernel_size=2, stride=2)`
    - 다시 가로/세로 크기를 절반으로 줄입니다.
    - `32 x 32`가 `16 x 16`이 됩니다.
    - 결과 shape은 `(batch_size, 32, 16, 16)`입니다.

6. `x.view(-1, 32 * 16 * 16)`
    - `nn.Linear`는 `(batch_size, features)` 형태의 2차원 입력을 받습니다.
    - 그래서 `(batch_size, 32, 16, 16)`을 `(batch_size, 8192)`로 펼칩니다.
    - 여기서 `8192 = 32 * 16 * 16`입니다.

따라서 첫 번째 fully connected layer는 아래처럼 작성합니다.

```python
self.fc1 = nn.Linear(32 * 16 * 16, 64)
```

정리하면 `32 * 16 * 16`은 두 번째 convolution과 두 번의 pooling을 거친 뒤 남은 feature map의 전체 원소 개수입니다.

## 1. CNN의 탄생: 도대체 왜 "합성곱"인가?

과거의 인공신경망은 이미지를 처리할 때 2차원 이미지를 1차원 선으로 길게 늘어뜨려 처리했지. (예: $64 \times 64 = 4096$ 길이의 벡터). 하지만 이렇게 하면 공간적 지역성(Spatial Locality)이 완전히 파괴된다네. 코 바로 옆에 눈이 있다는 2차원적 위치 정보가 사라지는 것이지.

**역사적 배경:**
1950년대 데이비드 허블(David Hubel)과 토르스텐 위젤(Torsten Wiesel)은 고양이의 시각 피질 세포가 시야의 '특정 국소 영역'에만 반응한다는 사실을 발견했네. 이를 바탕으로 1980년대 얀 르쿤(Yann LeCun) 교수는 이미지의 부분적인 특징(Edge, Corner 등)을 훑어가며 추출하는 **LeNet**을 제안했는데, 이것이 바로 CNN의 시초라네.

CNN은 전체를 한 번에 보지 않고, 돋보기(Kernel/Filter)를 들고 이미지를 훑으며(Sliding) 부분의 패턴을 찾아내는 아주 수학적이고 우아한 방식이지.

## 2. 텐서(Tensor)의 차원 변환 공식

네트워크를 지날 때 이미지의 가로세로 크기(Dimension)가 변하는 것은 철저히 아래의 **합성곱 출력 크기 공식**을 따른다네. 이 공식을 알면 어떤 레이어를 만나든 크기를 예측할 수 있지.

$$O = \left\lfloor \frac{W - K + 2P}{S} \right\rfloor + 1$$

* $O$: 출력 크기 (Output Size)
* $W$: 입력 크기 (Input Size, 여기서는 64)
* $K$: 커널(돋보기)의 크기 (Kernel Size)
* $P$: 패딩 (Padding, 이미지 테두리에 0을 덧대는 두께)
* $S$: 스트라이드 (Stride, 돋보기가 한 번에 이동하는 칸 수)
* $\lfloor \cdot \rfloor$: 내림(Floor) 연산

이제 이 공식을 머릿속에 넣고, 자네의 데이터가 네트워크를 통과하는 과정을 시각적인 덩어리(Volume)로 상상해 보세!

---

## 3. 직관적인 시각화: 데이터의 여정

1. **입력 데이터 (Input):** Shape: (3, 64, 64).
가로 64, 세로 64 크기의 픽셀 판이 3장(Red, Green, Blue) 겹쳐 있는 직육면체 블록을 상상하게. 이것이 자네가 모델에 던져준 원본 이미지라네.


2. **첫 번째 합성곱 (Conv1):** Shape: (16, 64, 64).
**`nn.Conv2d(3, 16, kernel_size=3, padding=1)`**

* **공식 적용:** $O = \lfloor \frac{64 - 3 + 2(1)}{1} \rfloor + 1 = 64$
* **시각화:** `padding=1` 덕분에 가로세로 64 크기가 그대로 유지되네. 하지만 16개의 서로 다른 돋보기(필터)가 이미지를 훑었기 때문에, 두께(채널)가 3장에서 16장으로 두꺼워진 64x64 블록이 탄생했네. 이제 이 블록은 색상이 아니라 '16종류의 이미지 특징(윤곽선 등)'을 담고 있지.


3. **첫 번째 풀링 (Pool1):** Shape: (16, 32, 32).
**`nn.MaxPool2d(kernel_size=2, stride=2)`**

* **공식 적용:** $O = \lfloor \frac{64 - 2 + 2(0)}{2} \rfloor + 1 = 32$
* **시각화:** 풀링은 정보의 '핵심'만 남기고 크기를 압축하는 과정이네. 2x2 영역마다 가장 강한 신호(Max) 1개만 뽑아내므로, 가로세로 길이가 정확히 반토막(32x32) 나게 되지. 두께(16)는 변하지 않네.


4. **두 번째 합성곱 (Conv2):** Shape: (32, 32, 32).
**`nn.Conv2d(16, 32, kernel_size=3, padding=1)`**

* **공식 적용:** $O = \lfloor \frac{32 - 3 + 2(1)}{1} \rfloor + 1 = 32$
* **시각화:** 이전 블록을 받아 다시 32개의 더 정교한 돋보기로 특징을 찾아내네. 면적은 32x32로 유지되지만, 두께(채널)는 32장으로 더욱 두꺼워졌네. 이제 눈, 코 같은 더 복잡한 형태를 인식하기 시작하는 단계지.


5. **두 번째 풀링 (Pool2):** Shape: (32, 16, 16).
**`nn.MaxPool2d(kernel_size=2, stride=2)`**

* **공식 적용:** $O = \lfloor \frac{32 - 2 + 2(0)}{2} \rfloor + 1 = 16$
* **시각화:** 다시 한번 핵심만 남기고 압축하네. 32x32 면적이 16x16으로 줄어들지. 이로써 우리는 가로 16, 세로 16, 두께 32인 단단한 '특징 블록'을 얻게 된 것이네.


6. **평탄화 (Flatten & View):** Shape: (8192).
**`x.view(-1, 32 * 16 * 16)`**

* **시각화:** 딥러닝의 마지막 판단(분류)을 내리는 `nn.Linear` 레이어는 3차원 블록을 소화하지 못하네. 그래서 32장 × 16칸 × 16칸으로 이루어진 이 정육면체 블록을 망치로 쳐서 **8,192개의 알갱이가 일렬로 늘어선 아주 긴 1차원 선**으로 펼쳐주는 것이지.

## 4. 핵심 수학적 정의 및 단축어 설명

### ReLU (Rectified Linear Unit)

코드에서 `F.relu()`가 쓰였지? 이 활성화 함수의 수학적 정의는 매우 단순하지만 혁명적이라네.


$$f(x) = \max(0, x)$$


이 함수는 음수 값은 0으로 날려버리고, 양수 값은 그대로 통과시키지. 인공신경망에 비선형성(Non-linearity)을 부여하여, 모델이 단순한 선형 방정식의 한계를 넘어 복잡한 곡선과 패턴을 학습할 수 있게 만들어준 1등 공신이라네.

### Max Pooling (최대 풀링)

풀링 연산의 목적은 이동 불변성(Translation Invariance)을 확보하는 것이네. 고양이가 사진의 왼쪽 위에 있든, 오른쪽 아래에 있든 동일하게 고양이로 인식해야 하니까!


$$y = \max_{i \in R} (x_i)$$


(여기서 $R$은 $2 \times 2$ 국소 영역)
위 수식처럼 주변 영역에서 가장 강한 자극(특징)만 가져오기 때문에, 약간의 픽셀 이동이 있어도 모델이 헷갈리지 않게 만들어 주며, 연산해야 할 파라미터(Parameter) 수를 극적으로 줄여주는 효과가 있지.

## 5. 응용 분야

자네가 작성한 이 SimpleCNN 구조는 기초적이지만 그 활용도는 무궁무진하다네.

* **의료 영상 분석:** X-ray나 MRI 사진을 $64 \times 64$ 또는 더 큰 크기로 입력하여 종양의 유무를 분류(Classification)할 수 있네.
* **제조업 결함 탐지:** 컨베이어 벨트를 지나가는 부품 사진을 찍어, 정상 제품과 스크래치가 있는 불량품을 구별하는 공장 자동화(Smart Factory)의 핵심 기술로 쓰이지.
* **자율주행:** 자동차 카메라가 앞차의 후미등이나 표지판을 인식하는 데에도 이런 합성곱 연산이 초당 수십 번씩 일어나고 있다네.

자, 텐서가 흘러가는 모습이 마치 건축물이 깎이고 다듬어지는 과정처럼 직관적으로 그려지지 않는가? 더 깊은 수렁으로 빠져보겠다면 다음 질문을 선택해 보게!

## 8. 모델 학습 및 검증 함수 구현 

```python
# 학습 함수 정의
def train(model, criterion, optimizer, train_loader):
    
    # 모드 설정
    model.train()
    
    # 초기화
    running_loss = 0.0
    corrects = 0
    
    # 배치 처리
    for images, labels in train_loader:
        images, labels = images.float(), labels.long() 
        images, labels = images.to(device), labels.to(device)
        
        # 역전파 및 파라미터 업데이트
        optimizer.zero_grad()# 역전파 전에 옵티마이저의 기울기를 0으로 초기화 
        outputs = model(images)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
        
        # 정확도 및 손실 추적
        running_loss += loss.item()
        _, preds = torch.max(outputs, 1)
        corrects += torch.sum(preds == labels.data)
        
    # 에포크 손실 및 정확도 계산
    epoch_loss = running_loss / len(train_loader)
    epoch_acc = corrects.double() / len(train_loader.dataset)
    
    return epoch_loss, epoch_acc
```

```python
# 검증 함수 정의
def validate(model, criterion, val_loader):
    
    model.eval()
    
    val_loss = 0.0
    val_corrects = 0
    
    with torch.no_grad():
        for images, labels in val_loader:
            images, labels = images.float(), labels.long()  # 데이터 타입 변환
            images, labels = images.to(device), labels.to(device)
            
            outputs = model(images)
            loss = criterion(outputs, labels)
            
            val_loss += loss.item()
            _, preds = torch.max(outputs, 1)
            val_corrects += torch.sum(preds == labels.data)
            
    epoch_loss = val_loss / len(val_loader)
    epoch_acc = val_corrects.double() / len(val_loader.dataset)
    return epoch_loss, epoch_acc
```

criterion = nn.CrossEntropyLoss()

# 성능 결과를 저장할 리스트
train_losses = []
train_accuracies = []
val_losses = []
val_accuracies = []

# 모델 학습 및 검증
num_epochs = 10
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model.to(device)

## 10. 모델 학습 및 검증 루프 구현 

```python
for epoch in range(num_epochs):
    
    train_loss, train_acc = train(model, criterion, optimizer, train_loader)
    val_loss, val_acc = validate(model, criterion, val_loader)
    
    train_losses.append(train_loss)
    train_accuracies.append(train_acc.item())
    val_losses.append(val_loss)
    val_accuracies.append(val_acc.item())
    
    print(f"Epoch {epoch+1}/{num_epochs}, "
          f"Train Loss: {train_loss:.4f}, "
          f"Train Accuracy: {train_acc:.4f}, "
          f"Validation Loss: {val_loss:.4f}, "
          f"Validation Accuracy: {val_acc:.4f}")
```
